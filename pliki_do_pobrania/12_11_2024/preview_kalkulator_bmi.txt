        function oblicz(){
            var waga = parseFloat(form.input_waga.value);
            var wzrost = parseFloat((form.input_wzrost.value)/100)
            var bmi = (waga/(wzrost*wzrost));
            bmi = bmi.toFixed(2);
            switch(true){
                case(bmi<16):
                label_wynik.innerHTML=bmi+'<p class="p_wyglodzenie">Wygłodzenie</p>'
                break;
                case(bmi>=16 && bmi<17):
                label_wynik.innerHTML=bmi+'<p class="p_wychudzenie">Wychudzenie</p>'
                break;
                case(bmi>=17 && bmi<18.5):
                label_wynik.innerHTML=bmi+'<p class="p_niedowaga">Niedowaga</p>'
                break;
                case(bmi>=18.5 && bmi<25):
                label_wynik.innerHTML=bmi+'<p class="p_waga_prawidlowa">Waga prawidłowa</p>'
                break;
                case(bmi>=25 && bmi<30):
                label_wynik.innerHTML=bmi+'<p class="p_nadwaga">Nadwaga</p>'
                break;
                case(bmi>=30 && bmi<35):
                label_wynik.innerHTML=bmi+'<p class="p_otylosc1">Otyłość I stopnia</p>'
                break;
                case(bmi>=35 && bmi<40):
                label_wynik.innerHTML=bmi+'<p class="p_otylosc2">Otyłość II stopnia</p>'
                break;
                case(bmi>=40):
                label_wynik.innerHTML=bmi+'<p class="p_otylosc3">Otyłość III stopnia</p>'
                break;
                default:
                label_wynik.innerHTML='<p class="p_error">Coś poszło nie tak.</p>'
            }
            console.log("bmi="+bmi);
            console.log("waga="+waga);
            console.log("wzrost="+wzrost);
        }

    </script>
