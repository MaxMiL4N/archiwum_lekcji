        // Wyświetl sumę wszystkich liczb ujemnych większych od -5. (Pętla wykonuje się 10 razy dla przykładu)
        var suma = 0;
        var licznik1 = 0;
        for (var i=-5;i<5;i++) {
            if (i<0) {
            suma+=i;
            }
            licznik1++;
        }

        // ===== ZADANIE 2 =====
        // Obliczanie silni;
        // Silnia !     5! = 1 * 2 * 3 * 4 * 5
        // Użytkownik podaje liczbę
        var silnia = parseInt(prompt("Podaj liczbę do obliczania silni:"));
        var wynik_silni = 1;
        for (var a=1;a <= silnia;a++) {
            wynik_silni*=a;
        }
        document.write(silnia,"! wynosi ",wynik_silni,"<hr>");

        // ===== ZADANIE 3 =====
        // Pętla wyświetlająca liczby w formie listy nienumerowanej (<ul>)

            document.write("<li>",i,"</li>");
        }
        document.write("</ul>");

        // Tabela
        document.write("<table> <tr>");
        for (var i = 1; i <= 5; i++) {
            document.write("<td>",i,"</td>");
