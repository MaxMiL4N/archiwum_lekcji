        function oblicz_raty() {
            var ilosc_rat = input_raty.value;
            var miasto = wybor_miasta.value;
            if (checkbox_react.checked==true && checkbox_js.checked==true && input_raty.value!="") {
                var calkowity_koszt_kursow=8000;
                var jedna_rata = calkowity_koszt_kursow/ilosc_rat;
                p_obliczone_raty.innerHTML="Kurs odbędzie się w "+miasto+". Koszt całkowity:"+calkowity_koszt_kursow+"zł. Płacisz"+ilosc_rat+"rat po"+jedna_rata+"zł."
            }
            else if (checkbox_react.checked==true && checkbox_js.checked==false && input_raty.value!="") {
                var calkowity_koszt_kursow=5000;
                var jedna_rata = calkowity_koszt_kursow/ilosc_rat;
                p_obliczone_raty.innerHTML="Kurs odbędzie się w "+miasto+". Koszt całkowity:"+calkowity_koszt_kursow+"zł. Płacisz"+ilosc_rat+"rat po"+jedna_rata+"zł."
            }
            else if (checkbox_react.checked==false && checkbox_js.checked==true && input_raty.value!="") {
                var calkowity_koszt_kursow=3000;
                var jedna_rata = calkowity_koszt_kursow/ilosc_rat;
                p_obliczone_raty.innerHTML="Kurs odbędzie się w "+miasto+". Koszt całkowity: "+calkowity_koszt_kursow+"zł. Płacisz "+ilosc_rat+" rat po "+jedna_rata+" zł."
            }
            else {
                p_obliczone_raty.innerHTML="Nie wybrano żadnego kursu lub nie podano ilości rat.";
            }
            
        }
    </script>
